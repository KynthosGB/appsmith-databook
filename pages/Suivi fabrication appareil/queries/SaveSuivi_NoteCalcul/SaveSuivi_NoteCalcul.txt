INSERT INTO public.workflow_note_calcul (
  groupe_appareil_id,
  numero,
  indice,
  date_objectif,
  date_edition,
	nombre_jours
)
VALUES (
  {{ this.params.groupe_appareil_id }},
  {{ this.params.numero }},
  {{ this.params.indice }},
  {{ this.params.date_objectif }},
  {{ this.params.date_edition }},
  {{ this.params.nombre_jours }}
)
ON CONFLICT (groupe_appareil_id) DO UPDATE SET
  numero        = EXCLUDED.numero,
  indice        = EXCLUDED.indice,
  date_objectif = EXCLUDED.date_objectif,
  date_edition  = EXCLUDED.date_edition,
  nombre_jours  = EXCLUDED.nombre_jours;
