WITH params AS (
  SELECT '{{ appsmith.URL.queryParams.numero_appareil }}'::text AS num_app
),

-- 1) suppression complète du sommaire existant de l'appareil
del_all AS (
  DELETE FROM public.sommaire_appareils
  WHERE numero_appareil = (SELECT num_app FROM params)
  RETURNING 1
),

-- 2) réinsertion complète depuis le sommaire maître
ins_all AS (
  INSERT INTO public.sommaire_appareils
         (numero_appareil, master_id, include, chapter_ord_override, sub_ord_override)
  SELECT
    p.num_app,
    s.master_id,
    TRUE,
    NULL,
    NULL
  FROM params p
  CROSS JOIN public.sommaire s
  RETURNING 1
),

-- 3) recalcul des ordres des chapitres niveau 1
new_orders AS (
  SELECT
    sa.numero_appareil,
    sa.master_id,
    ROW_NUMBER() OVER (
      PARTITION BY sa.numero_appareil
      ORDER BY s.default_chapter_ord
    ) AS new_ord
  FROM public.sommaire_appareils sa
  JOIN public.sommaire s
    ON s.master_id = sa.master_id
  JOIN params p
    ON p.num_app = sa.numero_appareil
  WHERE s.level = 1
),

-- 4) mise à jour des overrides d'ordre
upd_idx AS (
  UPDATE public.sommaire_appareils t
  SET chapter_ord_override = n.new_ord
  FROM new_orders n
  WHERE t.numero_appareil = n.numero_appareil
    AND t.master_id = n.master_id
  RETURNING 1
)

SELECT
  (SELECT COALESCE(COUNT(*), 0) FROM del_all) AS lignes_supprimees,
  (SELECT COALESCE(COUNT(*), 0) FROM ins_all) AS lignes_recreees,
  (SELECT COALESCE(COUNT(*), 0) FROM upd_idx) AS chapitres_reindexes;
