SELECT *
FROM liste_ccpu_nomenclature_appareil
WHERE numero_affaire = '{{ 
  appsmith.URL.queryParams.numero_affaire || DonneesGeneralesAffaire.data?.[0]?.numero_affaire 
}}'
  AND numero_appareil = '{{ 
  appsmith.URL.queryParams.numero_appareil || CaracteristiquesAppareils.data?.[0]?.numero_appareil 
}}'
ORDER BY code DESC;
