  groupe_appareil_id,
  preparateur_id,
  date_objectif,
  date_debut,
  date_fin,
	nombre_jours
)

  {{ this.params.preparateur_id }},
  {{ this.params.date_objectif }},
  {{ this.params.date_debut }},
  {{ this.params.date_fin }},

  preparateur_id   = EXCLUDED.preparateur_id,
  date_objectif = EXCLUDED.date_objectif,
  date_debut    = EXCLUDED.date_debut,
  date_fin      = EXCLUDED.date_fin,
  nombre_jours  = EXCLUDED.nombre_jours;
