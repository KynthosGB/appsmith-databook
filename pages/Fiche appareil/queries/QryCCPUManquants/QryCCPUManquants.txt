SELECT
  rep,
  srsm,
  code,
  CASE
    WHEN copied_at IS NULL THEN '⚠️Non traité'
    ELSE '✅Traité'
  END AS Statut
FROM ccpu_copies_manuelles
WHERE numero_affaire = {{ CaracteristiquesAppareils.data[0]?.numero_affaire }}
  AND nom_appareil   = {{ CaracteristiquesAppareils.data[0]?.nom_appareil }}
ORDER BY rep, srsm, code;
